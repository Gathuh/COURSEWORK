\newcommand{\parabola}[1]{
\begin{tikzpicture}
	\begin{axis}
		[
		xmin=-5,xmax=5
		ymin-1,ymax=20
		grid=major
		
		]
		\addplot[mark=none, thick, red, samples]{#1*x^2};
	\end{axis}
\end{tikzpicture}
}
